// backend/src/services/location.rs
use geo::Point;
use reverse_geocoder::{ReverseGeocoder, SearchResult};

pub async fn detect_country(lat: f64, lon: f64) -> Result<String> {
    let geocoder = ReverseGeocoder::new();
    let point = Point::new(lon, lat);
    
    let result = geocoder.search(point)?;
    
    // result.record.cc -> "PL", "DE", "UA", etc.
    Ok(result.record.cc.to_string())
}

// Використання
let country = detect_country(52.2297, 21.0122).await?; // "PL" (Польща)
let regulations = get_regulations_for_country(&country).await?;

### 2. Багатомовна база даних:

sql

```sql
-- База видів риби (багатомовна)
CREATE TABLE fish_species (
    id UUID PRIMARY KEY,
    scientific_name VARCHAR(100) UNIQUE,
    regions TEXT[], -- ['EU', 'UA', 'PL']
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE fish_names (
    id UUID PRIMARY KEY,
    fish_id UUID REFERENCES fish_species(id),
    language VARCHAR(5), -- 'uk', 'en', 'pl', 'de'
    common_name VARCHAR(100),
    UNIQUE(fish_id, language)
);

-- Приклад даних
INSERT INTO fish_species (scientific_name, regions) 
VALUES ('Cyprinus carpio', ARRAY['EU', 'UA', 'PL', 'DE']);

INSERT INTO fish_names (fish_id, language, common_name) VALUES
    ('...', 'uk', 'Короп'),
    ('...', 'en', 'Common Carp'),
    ('...', 'pl', 'Karp'),
    ('...', 'de', 'Karpfen');

-- Запит
SELECT fn.common_name, fs.scientific_name
FROM fish_species fs
JOIN fish_names fn ON fs.id = fn.fish_id
WHERE fn.language = 'pl' AND 'PL' = ANY(fs.regions);
```

let regional_baits = db.baits.get(&country)
