use sqlx::PgPool;
use anyhow::Result;
use std::collections::HashMap;

pub struct LocalizationService {
    pool: PgPool,
    cache: moka::future::Cache<String, HashMap<String, String>>,
}

impl LocalizationService {
    pub fn new(pool: PgPool) -> Self {
        let cache = moka::future::Cache::builder()
            .time_to_live(std::time::Duration::from_secs(3600))
            .build();
        
        Self { pool, cache }
    }
    
    /// ĞÑ‚Ñ€Ğ¸Ğ¼Ğ°Ñ‚Ğ¸ Ğ½Ğ°Ğ·Ğ²Ğ¸ Ñ€Ğ¸Ğ± Ğ½Ğ° Ğ¿ĞµĞ²Ğ½Ñ–Ğ¹ Ğ¼Ğ¾Ğ²Ñ–
    pub async fn get_fish_names(&self, language: &str) -> Result<HashMap<String, String>> {
        let cache_key = format!("fish_names_{}", language);
        
        if let Some(cached) = self.cache.get(&cache_key).await {
            return Ok(cached);
        }
        
        let names: Vec<(String, String)> = sqlx::query_as(
            "SELECT fs.scientific_name, fn.common_name 
             FROM fish_species fs 
             JOIN fish_names fn ON fs.id = fn.fish_id 
             WHERE fn.language = $1"
        )
        .bind(language)
        .fetch_all(&self.pool)
        .await?;
